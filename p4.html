Ejemplo de un password cracker for fuerza bruta.

1. Código del ejercicio (Python)

Brute Force SHA-256

Este script genera todas las combinaciones posibles de 4 letras y compara su hash SHA-256 con uno objetivo.


import hashlib
import string

lenght = 4

def product_gen(items, repeat):
    if repeat == 0:
        yield ()
        return

    for item in items:
        for combo in product_gen(items, repeat - 1):
            yield (item,) + combo

alpha = list(string.ascii_lowercase)

hash_guardado = '3cdea1b06e8f4c08213ca4cbd702a27871914c971ee226cf9debf88b8435c375'

for candidate in list(product_gen(alpha, repeat=lenght)):
    password = "".join(candidate)
    hash_resultado = hashlib.sha256(password.encode()).hexdigest()

    if hash_guardado == hash_resultado:
        print(password)
